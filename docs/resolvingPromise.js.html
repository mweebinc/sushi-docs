/**
 * Use promise as variable
 * @returns {Promise<unknown>}
 */
function resolvingPromise() {
    let res;
    let rej;
    const promise = new Promise((resolve, reject) => {
        res = resolve;
        rej = reject;
    });
    promise.resolve = res;
    promise.reject = rej;
    return promise;
}

export default resolvingPromise;
