/**
 * This is a global Application Configuration
 * @type {{MOODLE_URL: string, MOODLE_TOKEN: string}}
 */
const config = {
    MOODLE_URL: '',
    MOODLE_TOKEN: '',
}
const Config = {
    get: function (key) {
        if (config.hasOwnProperty(key)) {
            return config[key];
        }
        throw new Error('Configuration key not found.');
    },
    set: function (key, value) {
        config[key] = value
    }
}
module.exports = Config;
