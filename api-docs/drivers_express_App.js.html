const express = require('express');
const router = require('./routers');
const HttpServer = require('../http/HttpServer');
const Config = require('../../Config');
const getDatabaseController = require('../../controllers/database');
const getEmailController = require('../../controllers/email');
const SetupSchemaUseCase = require('../../domain/schema/usecases/SetupUseCase');

/**
 * an express app
 */
class App {
    constructor(options) {
        this.options = options;
        this.express = express();
        this.express.disable('x-powered-by');
        if (this.options.mountPath) {
            this.express.use(this.options.mountPath, router());
        } else {
            this.express.use(router());
        }
        const {databaseURI, databaseName} = this.options;
        this.databaseController = getDatabaseController(databaseURI, {dbName: databaseName});
        if (options.email) {
            this.emailController = getEmailController(options.email);
        }
        const dependency = {
            databaseController: this.databaseController,
            emailController: this.emailController,
        }
        Config.put(Object.assign({}, this.options, dependency));
    }

    initializeDatabase() {
        //save default schema
        const setup = new SetupSchemaUseCase(this.databaseController);
        return setup.execute()
            .catch((error) => {
                //ignore if already exist
            });
    }



    startHttpServer() {
        this.httpServer = new HttpServer(this.options.port, this.express);
        return this.httpServer.start()
            .then(() => console.log('rest server running...'));
    }

    start() {
        return Promise.resolve()
            .then(() => this.initializeDatabase())
            .then(() => this.startHttpServer());
    }

    stop() {
        return Promise.resolve()
            .then(() => this.httpServer.close());
    }
}

module.exports = App;
