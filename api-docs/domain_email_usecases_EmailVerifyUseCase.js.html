const ParseError = require('../../../ParseError');
const Auth = require('../../../Auth');
const Handlebars = require('handlebars');
const fs = require('fs');
const randomString = require('../../../randomString');
const EmailToken = require('../../../EmailToken');

/**
 * This class used to send verification to Email
 */
class EmailVerifyUseCase {
    constructor(config) {
        this.config = config;
        this.database = config.databaseController;
        this.emailController = config.emailController;
        this.master = Auth.getMaster(this.database);
        this.response = {};
    }

    sendEmail(email) {
        if (!this.emailController) return;
        const buffer = fs.readFileSync('src/drivers/express/views/verify-email.html');
        const template = Handlebars.compile(buffer.toString());
        const token = randomString(10);
        const link = this.config.verifyEmailSuccessURL + '/signup/' + token;
        EmailToken.put(token, true);
        const context = {
            link: link
        };
        const message = {
            to: email,
            subject: "入学申請・ユーザ情報登録承認【日本寿司リーディングアカデミー】",
            html: template(context),
        };
        return this.emailController.sendMail(message);
    }



    writeResponse() {
        return this.response;
    }

    validate(email) {
        if (!email) {
            throw new ParseError(ParseError.EMAIL_MISSING, 'email is required.');
        }
    }

    execute(auth, params) {
        return Promise.resolve()
            .then(() => this.validate(params.email))
            .then(() => this.sendEmail(params.email))
            .then(() => this.writeResponse());
    }
}

module.exports = EmailVerifyUseCase;
