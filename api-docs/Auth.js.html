const ParseError = require('./ParseError');


const master = null;

/**
 * Auth tells you who is requesting
 */
class Auth {
    /**
     * @param database
     * @param isMaster
     * @param user
     */
    constructor(database, isMaster, user) {
        this.database = database;
        this.isMaster = isMaster;
        this.user = user;
        this.acl = this.isMaster ? [master] : ['*'];
    }

    /**
     * Get the Role of the user
     * @returns {PromiseLike<any> | Promise<any>}
     */
    findRole() {
        if (this.isMaster || !this.user) return;
        const query = {users: [this.user]};
        return this.database.find([master], 'roles', query)
            .then(roles => {
                this.acl = this.acl.concat(roles.map(r => 'role:' + r.name.toLowerCase()));
            });
    }


    getAccess() {
        this.user && this.acl.push(this.user.id.toLowerCase());
    }

    getAcl() {
        return Promise.resolve()
            .then(() => this.getAccess())
            .then(() => this.findRole())
            .then(() => this.acl);
    }
}

function getMaster(database) {
    return new Auth(database, true);
}

/**
 * Get Authentication By Session Token
 * @param database
 * @param token
 * @param masterKey
 * @returns {PromiseLike<Auth> | Promise<Auth>}
 */
function getAuthBySession(database, token, masterKey) {
    const query = {token};
    return database.find([master], 'sessions', query)
        .then((results) => {
            if (results.length !== 1) {
                throw new ParseError(ParseError.INVALID_SESSION_TOKEN, 'Invalid session token');
            }
            const session = results[0];
            const user = session.user;
            const now = new Date();
            const expiresAt = session.expiresAt;
            // if (now > expiresAt) {
            //     throw new ParseError(ParseError.INVALID_SESSION_TOKEN, 'Session token is expired.');
            // }
            return new Auth(database, session.masterKey === masterKey, user);
        });
}

Auth.getMaster = getMaster;
Auth.getAuthBySession = getAuthBySession;

module.exports = Auth;
