const http = require('http');
const https = require('https');

const {URL} = require('url');
const clients = {
    'http:': http,
    'https:': https,
};

class HTTPResponse {
    constructor(response, body) {
        this.status = response.statusCode;
        this.headers = response.headers || {};
        this.body = body;
        this.ok = this.status >= 200 && this.status < 300;
    }

    text() {
        if (Buffer.isBuffer(this.body)) {
            return this.body.toString();
        }
    }

    json() {
        if (Buffer.isBuffer(this.body)) {
            const text = this.body.toString();
            try {
                return JSON.parse(text);
            } catch (e) {
            }
            return text;
        }
    }

}


function callback(resolve, reject) {
    return function (res) {
        const chunks = [];
        res.on('data', chunk => {
            chunks.push(chunk);
        });
        res.on('end', () => {
            const body = Buffer.concat(chunks);
            const response = new HTTPResponse(res, body);
            resolve(response);
        });
        res.on('error', error => {
            reject(error);
        });
    };
}

/**
 * Use this function for external request
 * @param host
 * @param options
 * @returns {Promise<unknown>}
 */
function fetch(host, options = {}) {
    const url = new URL(host);
    const client = clients[url.protocol];
    return new Promise((resolve, reject) => {
        const req = client.request(url, options, callback(resolve, reject));
        options.body && req.write(options.body);
        req.on('error', error => {
            reject(error);
        });
        req.end();
    });
}

module.exports = fetch;
