import changes from "../../changes";

/**
 * This class handle Change Email Page
 */
class ChangeEmailPresenter{
    /**
     * @param view
     * @param updateObjectUseCase
     */
    constructor(view, updateObjectUseCase) {
        this.view = view;
        this.updateObjectUseCase = updateObjectUseCase;
    }

    /**
     * Initialize user data
     */
    init(){
        this.user = this.view.getCurrentUser()
    }
    goBack(){
        this.view.navigateBack();
    }

    /**
     * Update email
     * @param email
     * @param confirmEmail
     */
    updateEmail({email, confirmEmail}){
        if(email !== confirmEmail){
            this.view.showError(this.view.t("email_must_be_the_same"),this.view.t("change_email_fail"),this.view.t("ok"));
            return;
        }
        const current = this.view.getCurrentUser();
        const user = {
            id: current.id,
            username: email,
            email,
            active: current.active
        }
        const action = this.updateObjectUseCase;
        Promise.resolve()
            .then(()=> action.execute('users', user))
            .then(()=>{
                this.view.showSuccess(this.view.t('email_address_has_been_changed'), this.view.t('email_address_change_completed'),this.view.t("ok"));
                this.goBack();
            })
            .catch(error => {
                if (error.code === 209){
                    this.view.showError(this.view.t('invalid_token'), this.view.t('error'));
                    this.view.navigateTo('/signin');
                    return;
                }
                this.view.showError(error);
            })
    }
}

export default ChangeEmailPresenter;
