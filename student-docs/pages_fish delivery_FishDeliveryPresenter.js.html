/**
 * This class handle the FishDelivery Page
 */
class FishDeliveryPresenter {
    /**
     * @param view
     * @param findObjectUseCase
     */
    constructor(view, findObjectUseCase) {
        this.view = view;
        this.findObjectUseCase = findObjectUseCase;
        this.limit = 10;
        this.current = 1;
        this.where = {};
        this.documents = [];
        this.progress = false;
    }

    /**
     * Initialize the data
     */
    init() {
        this.getData();
    }

    /**
     * Get the Shipping
     * @returns {*}
     */
    getData() {
        //get current user and retrieve only filter deliveries for user using email.
        const user = this.view.getUser();
        this.where = {user : {id : user.id}};
        const skip = (this.current - 1) * this.limit;
        const query = {count: true, limit: this.limit, skip, where: this.where, include: ['all']};
        this.setProgress(true);
        return this.findObjectUseCase.execute('shippings', query)
            .then(({count, objects}) => {
                this.documents = this.documents.concat(objects);
                this.view.setMore(count > this.documents.length);
                this.view.setDeliveries(this.documents);
                this.setProgress(false);
            })
            .catch(error => {
                this.setProgress(false);
                this.view.showError(error);
            });
    }

    setProgress(progress) {
        this.progress = progress;
        if (progress) {
            this.view.showProgress();
        } else {
            this.view.hideProgress();
        }
    }

    loadMore() {
        if (!this.progress) {
            this.current++;
            this.getData();
        }
    }

    onButtonClick(index){
        const document = this.documents[index];
        this.view.navigateTo('/shippings/' + document.id);
    }


}

export default FishDeliveryPresenter;
