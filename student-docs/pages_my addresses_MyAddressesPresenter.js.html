import {updateObjectUseCase} from "../../domain/object";

/**
 * This class handle My Address Page
 */
class MyAddressesPresenter {
    /**
     * @param view
     * @param findObjectUseCase
     * @param updateObjectUseCase
     */
    constructor(view, findObjectUseCase, updateObjectUseCase) {
        this.view = view;
        this.findObjectUseCase = findObjectUseCase;
        this.updateObjectUseCase = updateObjectUseCase;
        this.recipients = [];

    }

    /**
     * Initialize data
     */
    init() {
        this.user = this.view.getCurrentUser();
        this.setRecipients();
    }

    /**
     * Set the recipients
     */
    setRecipients() {
        const id = this.user.id;
        const query = {include: ['all'], where: {id}};
        this.view.showProgress();
        this.findObjectUseCase.execute('users', query)
            .then(([result]) => {
                this.recipients = this.recipients.concat(result.recipients)
                this.view.setRecipients(this.recipients);
                this.view.hideProgress();
            })
    }

    /**
     * When user click back button
     */
    backClick() {
        this.view.navigateBack();
    }

    /**
     * When user navigate to address
     */
    navigateFormAddress() {
        this.view.navigateTo("/my-addresses/form");
    }
}

export default MyAddressesPresenter;
