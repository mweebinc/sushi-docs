import React from 'react';
import {useTranslation} from "react-i18next";
import Context from './AppContext';
/**
 * AppProvider is the main Component store global state
 * @param children
 * @returns {JSX.Element}
 * @constructor
 */
function AppProvider({children}) {
    const [state, setState] = React.useState();
    const {i18n, t} = useTranslation(["common"]);

    function setGlobalState(_state) {
        setState({...state, ..._state});
    }

    function setLanguage(lng) {
        i18n.changeLanguage(lng);
    }

    function getValue() {
        return {
            translation: t,
            setLanguage: setLanguage.bind(this),
            ...state,
            setGlobalState: setGlobalState.bind(this),
        };
    }
    return <Context.Provider value={getValue()}>{children}</Context.Provider>;
}
export default AppProvider;
