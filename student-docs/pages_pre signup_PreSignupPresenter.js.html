/**
 * This class handle Pre Sign up Page
 */
class PreSignupPresenter{
    /**
     * @param view
     * @param emailUseCase
     */
    constructor(view, emailUseCase) {
        this.view = view;
        this.emailUseCase = emailUseCase;
    }

    /**
     * When the user submit the form
     */
    onSubmitForm(){
        const email = this.view.getEmail();

        this.emailUseCase.execute(email.email)
            .then(() => {
                this.view.showSuccess(this.view.t('you_have_just_been_sent'), this.view.t("check_your_email_to_continue_registration"), this.view.t("ok"))
                this.view.navigateTo('/signin');
            })
            .catch(() => {
                this.view.showError(this.view.t('an_error_occurred_link_not_sent'), this.view.t('error'));
            })

}
export default PreSignupPresenter;
