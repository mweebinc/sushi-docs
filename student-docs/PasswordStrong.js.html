/**
 * Check the validity of the password
 * @param password
 * @returns {string}
 */
function passwordStrong(password){
    const strongRegex = new RegExp("^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*])(?=.{8,20}$)");
    const oneUpperCase = new RegExp("^(?=.*[A-Z])");
    const oneLowerCase = new RegExp("^(?=.*[a-z])");
    const oneMoreNumber = new RegExp("^(?=.*[0-9])");
    const oneMoreSymbols = new RegExp("^(?=.*[!@#\$%\^&\*])");
    const limit = new RegExp("^(?=.{8,20}$)");
    const regexErrorMessage = [];

    if(!oneUpperCase.test(password))
        regexErrorMessage.push("at_least_one_uppercase_letter")
    if(!oneLowerCase.test(password))
        regexErrorMessage.push("one_or_more_lower_case_letters")
    if(!oneMoreNumber.test(password))
        regexErrorMessage.push("one_or_more_numbers")
    if(!oneMoreSymbols.test(password))
        regexErrorMessage.push("one_or_more_symbols")
    if(!limit.test(password))
        regexErrorMessage.push("at_least_8_and_no_more_than_20_characters_in_total");

    return !strongRegex.test(password) ? regexErrorMessage : undefined;

}
export default passwordStrong;
