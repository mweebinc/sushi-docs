/**
 * This class handle the Main Page
 */
class MainPresenter{
    /**
     * @param view
     * @param getCurrentUser
     * @param signOut
     * @param getAllSchema
     * @param getRolesByUserUseCase
     */
    constructor(view, {getCurrentUser, signOut, getAllSchema, getRolesByUserUseCase}) {
        this.view = view;
        this.getCurrentUserUseCase = getCurrentUser;
        this.signOutUseCase = signOut;
        this.getAllSchemaUseCase = getAllSchema
        this.getRolesByUserUseCase = getRolesByUserUseCase
    }


    componentDidMount(){
        this.init();
    }

    /**
     * Initialize the data
     */
    init(){
        this.view.showProgress();
        Promise.resolve()
            .then(() => this.getUser())
            .then(()=> this.getSchemas())
            .then(() => this.view.hideProgress())
            .catch(error => {
                this.view.hideProgress();
                if(error.code === 209){
                    this.view.navigateTo('/signin');
                }else{
                    this.view.showError(error);
                }
            })
    }

    /**
     * When user click sign-out button
     */
    signOutClick() {
        const options = {
            title: this.view.t('confirm'),
            message: this.view.t('are_you_sure_you_want_to_sign_out'),
            positiveButton: this.view.t("sign_out"),
            negativeButton: this.view.t("cancelB")
        }
        this.view.showDialog(options)
            .then(() => this.signOutUseCase.execute())
            .then(() => {
                this.view.navigateTo('/signin');
                this.view.reload();
            })
            .catch(error => {
                this.view.showError(error);
            });
    }

    getSchemas(){
        return this.getAllSchemaUseCase.execute()
            .then(schemas => {
                this.view.setSchemas(schemas)
            })
    }

    /**
     * Get the current user
     * @returns {*}
     */
    getUser(){
        return this.getCurrentUserUseCase.execute()
            .then(user => {
                this.user = user;
                this.view.setCurrentUser(user)
            })
    }

}
export default MainPresenter;
