/**
 * This class handle Shipping Details
 */
class ShippingDetailsPresenter {
    /**
     * @param view
     * @param getObjectUseCase
     */
    constructor(view, getObjectUseCase) {
        this.view = view;
        this.getObjectUseCase = getObjectUseCase;
    }

    /**
     * Initialize the data
     */
    init() {
        this.getData();
    }

    /**
     * When user want to navigate to change address
     */
    onClick() {
        this.view.navigateTo('/shippings/' + this.id + '/change-address')
    }

    /**
     * When user click back button
     */

    onBackClick() {
        this.view.navigateTo('/fish-delivery');
    }

    /**
     * Get the data
     */
    getData() {
        const id = this.view.getShippingId();
        this.id = id;
        this.view.showProgress();
        this.getObjectUseCase.execute('shippings', id)
            .then(result => {
                this.view.hideProgress();
                this.view.setObject(Object.assign({}, result));
            })
            .catch(error => {
                console.log(error);
            })
    }


}

export default ShippingDetailsPresenter;
